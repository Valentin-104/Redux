개요: 각각 최상단/최하단으로 스크롤하는 버튼을 만든다.

로직: 최상단으로 스크롤하는 경우, **window 객체**를 이용한다. 최하단으로 스크롤하는 경우, **useRef**를 호출하여 이를 최하단 빈 `<div>`의의 `ref`속성에 걸어주고, `scrollIntoView`메서드를 활용한다.

[Scroll to Top/Bottom](https://github.com/Valentin1495/React-Projects-for-Interview/tree/main/src/components/Scroll-Top-Bottom)

```tsx:src/components/Scroll-Top-Bottom/scroll-top-bottom.tsx
import { useEffect, useRef, useState } from 'react';
import './styles.css';

type Album = {
  id: number;
  title: string;
};

export default function ScrollTopBottom() {
  const [albums, setAlbums] = useState<Album[]>([]);
  const [loading, setLoading] = useState<boolean>(false);
  const [error, setError] = useState<string | null>(null);
  const ref = useRef<HTMLDivElement>(null);
  const scrollToTop = () => {
    window.scrollTo({
      top: 0,
      behavior: 'smooth',
    });
  };

  const scrollToBottom = () => {
    ref.current?.scrollIntoView({
      behavior: 'smooth',
    });
  };

  useEffect(() => {
    const getAlbums = async () => {
      try {
        setLoading(true);
        const response = await fetch(
          import.meta.env.VITE_JSON_PLACEHOLDER_API_URL + '/albums'
        );

        const data = await response.json();
        setAlbums(data);
      } catch (error) {
        setError(error instanceof Error ? error.message : String(error));
      } finally {
        setLoading(false);
      }
    };

    getAlbums();
  }, []);

  if (loading) return <p>Loading...</p>;
  if (error) return <p>{error}</p>;
  return (
    <div
      style={{
        padding: '50px',
      }}
    >
      <button className='scroll-btn' onClick={scrollToBottom}>
        Scroll to bottom
      </button>
      {albums.map(({ id, title }) => (
        <p key={id}>{title}</p>
      ))}
      <button className='scroll-btn' onClick={scrollToTop}>
        Scroll to top
      </button>
      <div ref={ref}></div>
    </div>
  );
}
```
